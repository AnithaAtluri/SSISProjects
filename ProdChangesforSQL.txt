SELECT
	ISNULL(RTRIM(STUFF (CaseNo, PATINDEX( '%[0-9]%', CaseNo), 0, '-')) 
	+ LEFT('        ', (10-LEN(RTRIM(STUFF (CaseNo, PATINDEX( '%[0-9]%', CaseNo), 0, '-'))))), '        ')
	+ '|' + ISNULL(RTRIM(SID) + LEFT('        ', (8-LEN(RTRIM(SID)))), '        ')
	+ '|' + ISNULL(RTRIM(TRN) + LEFT('          ', (10-LEN(RTRIM(TRN)))), '          ')
	+ '|' + ISNULL(RTRIM(TRS) + LEFT('    ', (4-LEN(RTRIM(TRS)))), '    ')
	AS AIS_Data
FROM MF_Court.dbo.tblCJIS
WHERE SID_Source = 'AIS'
	OR TRN_Source = 'AIS'
	OR TRS_Source = 'AIS'
